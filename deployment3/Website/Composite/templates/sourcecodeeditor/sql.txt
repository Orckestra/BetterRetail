-- simple select example

SELECT * FROM books	
	WHERE price > 100.00 and price < 150.00	
	ORDER BY title
	
SELECT books.title, count(*) AS Authors	
	FROM books	
	JOIN book_authors 		
		ON books.book_number = book_authors.book_number	
	GROUP BY books.title
